        * @dev Emits the ApprovalForAll event. The contract MUST allow
        *  multiple operators per owner.
        * @param _operator Address to add to the set of authorized operators
        * @param _approved True if the operator is approved, false to revoke approval
        */
    /// @notice postcondition isApprovedForAll(msg.sender, _operator) == _approved
    function setApprovalForAll(address _operator, bool _approved) external;

    /**
        * @notice Query if an address is an authorized operator for another address
        * @param _owner The address that owns the NFTs
        * @param _operator The address that acts on behalf of the owner
        * @return True if `_operator` is an approved operator for `_owner`, false otherwise
        */
    /// @notice postcondition return == _operatorApprovals[_owner][_operator]
    function isApprovedForAll(address _owner, address _operator) external view returns (bool);

    /// @notice postcondition _ownedTokensCount[_from] == __verifier_old_uint(_ownedTokensCount[_from]) - 1
    /// @notice postcondition _ownedTokensCount[_to] == __verifier_old_uint(_ownedTokensCount[_to]) + 1
    function transferFrom(address _from, address _to, uint256 _tokenId) external;

        * @param _from The current owner of the NFT
        * @param _to The new owner
        * @param _tokenId The NFT to transfer

    /// @notice postcondition _tokenOwner[_tokenId] == _to && _tokenOwner[_tokenId] != _from
    /// @notice postcondition _ownedTokensCount[_from] == __verifier_old_uint(_ownedTokensCount[_from]) - 1
    /// @notice postcondition _ownedTokensCount[_to] == __verifier_old_uint(_ownedTokensCount[_to]) + 1
    function safeTransferFrom(address _from, address _to, uint256 _tokenId) external;

    /**
        * @notice Transfers the ownership of an NFT from one address to another address
        * @dev Throws unless `msg.sender` is the current owner, an authorized

        * @param data Additional data with no specified format, sent in call to `_to`

    /// @notice postcondition _tokenOwner[_tokenId] == _to && _tokenOwner[_tokenId] != _from
    /// @notice postcondition _ownedTokensCount[_from] == __verifier_old_uint(_ownedTokensCount[_from]) - 1
    /// @notice postcondition _ownedTokensCount[_to] == __verifier_old_uint(_ownedTokensCount[_to]) + 1
    function safeTransferFrom(address _from, address _to, uint256 _tokenId, bytes calldata _data) external;
}

### Explanation of Postconditions:
1. **balanceOf**: The returned balance must equal the count of NFTs owned by the specified address.
2. **ownerOf**: The returned owner must match the stored owner for the given token ID and must not be the zero address.
3. **approve**: The approval for the token ID must either match the new approved address or be reset.
4. **getApproved**: The return value must be the approved address for the token ID.
5. **setApprovalForAll**: The approval status for the operator must reflect the new approval state.
6. **isApprovedForAll**: The return value must accurately reflect whether the operator is approved for the owner.
7. **transferFrom**: The token must be transferred correctly from the current owner to the new owner, and the new owner must not be the zero address. The count of owned tokens for both the sender and receiver must update correctly.
8. **safeTransferFrom**: This function must ensure the token is transferred correctly and the new owner is valid, with the same token count updates as `transferFrom`.
9. **safeTransferFrom with data**: Similar to the previous safeTransferFrom, ensuring that the transfer is valid and the token counts are updated correctly.

These postconditions ensure compliance with the expected behavior as defined in the ERC-721 standard while avoiding any syntax issues that could lead to compilation errors.
